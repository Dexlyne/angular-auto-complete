Activate on Focus

HTML
<body ng-app="mainApp">
    <div ng-controller="MainCtrl as ctrl">
        <input type="text"
            ng-model="ctrl.breakfast"
            placeholder="Breakfast"
            auto-complete="ctrl.autoCompleteOptions" />
    </div>
</body>
JavaScript
angular
    .module('mainApp', ['autoCompleteModule'])
    .controller('MainCtrl', function() {
        var that = this;
        that.breakfast = null;

        that.autoCompleteOptions = {
            minimumChars: 0,
            activateOnFocus: true,
            data: function (searchText) {
                searchText = searchText.toUpperCase();

                return _.filter(BREAKFAST, function (breakfast) {
                    return breakfast.startsWith(searchText);
                });
            }
        }
    });
