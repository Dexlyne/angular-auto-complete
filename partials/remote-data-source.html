Example 3. Remote Data Source

HTML
<body ng-app="mainApp">
    <div ng-controller="MainCtrl as vm">
        <input type="text" 
            ng-model="vm.stateName"   
            placeholder="State Name"
            auto-complete="vm.autoCompleteOptions" />
    </div>
</body>
JavaScript
angular
    .module('mainApp', ['autoComplete'])
    .controller('MainCtrl', MainCtrl);
    
    MainCtrl.$inject = ['$http'];
    function MainCtrl($http) {
        var that = this;
        that.stateName = null;
        
        that.autoCompleteOptions = {
            minimumChars: 1,
            data: function (term) {
                return $http.get('states.txt')
                    .then(function (response) {
                        // ideally filtering should be done on server
                        term = term.toUpperCase();
                        var match = _.filter(response.data, function (value) {
                            return value.name.startsWith(term);
                        });
                        return _.pluck(match, 'name');
                    });
            }
        }
    }
