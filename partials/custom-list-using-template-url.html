Custom List Using itemTemplateUrl

{{vm.selectedColor | json}}
HTML
<body ng-app="mainApp">
    <div ng-controller="MainCtrl as vm">
        <input type="text"
            ng-model="vm.colorName"
            style="width: 400px;"
            placeholder="Hex Color Code. Try 'a' or '0'"
            auto-complete="vm.autoCompleteOptions" />
    </div>
</body>
JavaScript
angular
    .module('mainApp', ['autoCompleteModule'])
    .controller('MainCtrl', function() {
        var that = this;
        that.colorName = null;
        that.selectedColor = null;

        that.autoCompleteOptions = {
            minimumChars: 1,
            data: function (term) {
                if (!term.startsWith('#')) {
                    term = '#' + term;
                }
                term = term.toUpperCase();
                return _.filter(CSS_COLORS, function (color) {
                    return color.code.startsWith(term);
                });
            },
            containerCssClass: 'color-codes',
            selectedTextAttr: 'code',
            itemTemplateUrl: 'templates/color-list-item.tpl.html',
            itemSelected: function (e) {
                that.selectedColor = e.item;
            }
        }
    });
