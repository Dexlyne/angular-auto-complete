HTML
<body ng-app="mainApp">
    <div ng-controller="MainCtrl as vm">
        <input type="text" 
            ng-model="vm.colorName"   
            placeholder="Color Name"
            auto-complete="vm.autoCompleteOptions" />
    </div>
</body>
JavaScript
angular
    .module('mainApp', ['autoComplete'])
    .controller('MainCtrl', function() {
        var that = this;
        that.colorName = null;

        that.autoCompleteOptions = {
            minimumChars: 1,
            data: function (term) {
                term = term.toUpperCase();
                return _.filter(CSS_COLORS, function (value) {
                    return value.name.startsWith(term);
                });
            },
            dropdownWidth: '400px',
            containerCssClass: 'color-codes',
            selectedTextAttr: 'name',
            itemTemplateUrl : 'templates/color-list-item.tpl.html',
            itemSelected: function (item) {
                that.selectedColor = item;
            }
        }
    });
